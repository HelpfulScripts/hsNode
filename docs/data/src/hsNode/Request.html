Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET and POST. The module wraps the respective `Node` calls in async/await patterns

   3 * and provides support for

   4 * - simple authentication methods, 

   5 * - local caching,  

   6 * - decoding of received comntent.

   7 * - request pacing

   8 * 

   9 * ### Authentication

  10 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

  11 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  12 * The following authentication schemes are currently supported:

  13 * - Basic

  14 * - Digest

  15 * 

  16 * ### Caching

  17 * If caching is enabled, `GET` requests will attempt to return an available cached content

  18 * before issuing the request to the server.

  19 * Caching is disabled by default and can be enabled by setting a caching directory location before any `GET` call

  20 * `request.cache = './data'`.

  21 * Once set, individual `request.get(url)` requests will by default use caching. This can be 

  22 * disabled on a per call basis by providing `false` as a second parameter:

  23 * `request.get(url, false)` behaves as if `cache` is undefined

  24 * 

  25 * ### Decoding

  26 * Content received from a server or a cache can be decoded before returning the result. 

  27 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  28 * `request.decode = Request.decoders.str2json`

  29 * 

  30 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  31 * - str2json applies JSON.parse to the content to return an object literal

  32 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  33 * 

  34 * ### Pacing

  35 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  36 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  37 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  38 * 

  39 * ### Usage:

  40 * ```

  41 * import { Request } from 'hsnode';

  42 * const request = new Request.Request();

  43 * request.cache = __dirname + '/../_data/cache';

  44 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  45 * request.setAuthToken(apptokens.token);

  46 * const content = await request.get('http://...');

  47 * ```

  48 */

 202    /**

 203     * issues a `GET` or `POST` call and returns the `url` response. 

 204     * If `usedCache` is `true`, the cached version will be returned if available. 

 205     * Otherwise an `http` or `https` request is initiated, the result cached and returned.

 206     * If `decode` is defined it is called with the retrieved data before returning the result. 

 207     * @param url the url to retrieve

 208     * @param method 'GET' or 'POST'

 209     * @param useCached if `true`, attempt to return a cahced version. If `false`, or no cached version

 210     * exists, a remote call will be attempted.

 211     * @param postData data to submit for a `POST` call

 212     */

 213    protected async decodedRequest(url:string|URL, method:'GET'|'POST', useCached:boolean, postData?:any):Promise {

 214        const options = this.getOptions(url, method);

 215        try { 

 216            const result = await this.requestOptions(options, useCached, postData); 

 217            return this.decode? this.decode(result, options) : result;

 218        }

 219        catch(e) { return undefined; }

 220    }

 230    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise {

 231        const fname = this.cache? this.cacheName(options) : undefined;

 232        if (fname && useCached && options.method === 'GET') { 

 233            try { return await this.readCached(fname); }

 234            catch(e) {} // no cache available

 235        }

 236

 237        let response: Response;

 238        const err = <{statusCode:string, statusMessage:string, url:string}>{};

 239        try { 

 240            if (this.pace) { 

 241                log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) requesting ${options.url}`); 

 242                response = await this.pace.add(() => this.request(options, postData));

 243                log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()})             `);

 244            }

 245            else { 

 246                log.info(`requesting ${options.url}`); 

 247                response = await this.request(options, postData);

 248            }

 249            if((response.response.statusCode||response.response.status) === 200) {

 250                if (fname && options.method === 'GET') {

 251                    try { await this.writeCached(fname, response.data); }

 252                    catch(e) { log.warn(`writing cache for ${fname}: ${e}`); }

 253                }

 254                return response.data;

 255            }

 256            log.warn(`${response.response.statusCode||response.response.status} requesting ${options.url}: ${response.response.statusMessage}`);

 257            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 258            err.statusMessage = response.response.statusMessage || 'no message';

 259        }

 260        catch(e) {

 261            log.warn(`error requesting ${options.url}: ${e.error.code}\n${log.inspect(e, 2)}`);

 262            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 263            err.statusMessage = JSON.stringify(e);

 264        }

 265        err.url = options.url;  

 266        fs.writeTextFile(fname + '.error.txt', JSON.stringify(err)); 

 267        throw(`response code ${err.statusCode} for ${options.url}`);

 268    }

 270    protected async request(options:Options, postData?:any):Promise {

 271        const httpProt:string = options.protocol.slice(0,-1);

 272        const http = protocol[httpProt];

 273        let auth = (options.headers && options.headers.Authorization);

 274        const response = await new Promise((resolve:(out:Response)=>void, reject:(e:Response)=>void) => {

 275            let data = ''; 

 276            log.debug(`requesting ${log.inspect(options, 4)}`);

 277            const req = http.request(options, (res:any) => {

 278                const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';

 279                log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);

 280                res.setEncoding(encoding);

 281                res.on('data', (chunk:string) => data += chunk);

 282                res.on('end', () => resolve({data:data, response:res}));

 283            });

 284            req.on('error', (e:any) => reject({data:'', error:e}));

 285    

 286            // write data to request body

 287            if (postData !== undefined) { req.write(postData); }

 288            req.end();

 289        });

 290        const prot = response.response.headers['www-authenticate'];

 291        if (this.credentials && prot) { 

 292            if (prot.trim().indexOf('Digest') === 0) {

 293                const digest = new Digest(this.credentials.user, this.credentials.password);

 294                digest.testDigestAuth(options, response.data, response.response);

 295                return this.request(options);

 296            }

 297            if (prot.trim().indexOf('Basic') === 0) {

 298                options.headers.Authorization = 'Basic ' + _btoa(`${this.credentials.user}:${this.credentials.password}`);

 299                return this.request(options);

 300            }

 301        } else {

 302            return response; 

 303        } 

 304    }

 339 * Implements a Digest authentication, used in {@link request `request`} call.

 341class Digest {

 342    nc = 0;

 343    username:string;

 344    password:string;

 345

 346    constructor(username:string, password:string) {

 347        this.username = username;

 348        this.password = password;

 349    }

 351    /**

 352     * Update and zero pad nc

 353     */

 354    updateNC():string {

 355        let max = 99999999;

 356        let padding = new Array(8).join('0') + '';

 357        this.nc = (this.nc > max ? 1 : this.nc + 1);

 358        let nc = this.nc + '';

 359        return padding.substr(0, 8 - nc.length) + nc;

 360    }

 362    /**

 363     * Parse challenge digest

 364     * @param qop 

 365     */

 366    generateCNONCE(qop:string) {

 367        let cnonce:any;

 368        let nc:string;

 369

 370        if (typeof qop === 'string') {

 371            let cnonceHash = createHash('md5');

 372            cnonceHash.update(Math.random().toString(36));

 373            cnonce = cnonceHash.digest('hex').substr(0, 16);

 374            nc = this.updateNC();

 375        }

 376        return {cnonce: cnonce, nc: nc};

 377    }

 379    /**

 380     * 

 381     * @param options 

 382     * @param data 

 383     * @param response 

 384     */

 385    testDigestAuth(options:Options, data:string, response:IncomingMessage): Options {

 386        log.debug(`received www-authenticate request for ${options.host}`);

 387

 388        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

 389        let ha1 = createHash('md5');

 390        let _str = `${this.username}:${challenge.realm}:${this.password}`;

 391        ha1.update(_str);

 392        let ha2 = createHash('md5');

 393        _str = `${options.method}:${options.path}`;

 394        ha2.update(_str);

 395    

 396        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

 397        let hash = createHash('md5');

 398        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

 399        hash.update(_str);

 400    

 401        // Setup response parameters

 402        let authParams:any = {

 403            realm: challenge.realm,

 404            username: this.username,

 405            uri: options.path,

 406            qop: challenge.qop,

 407            nonce: challenge.nonce,

 408            algorithm: "MD5",

 409            response: hash.digest('hex')

 410        };

 411    

 412        authParams = omitNull(authParams);

 413    

 414        if (cnonce) {

 415            authParams.nc = nc;

 416            authParams.cnonce = cnonce;

 417        }

 418    

 419        options.headers.Authorization = compileParams(authParams);

 420        return options;

 421    }

 426function omitNull(data:any) {

 427    let newObject = {};

 428    Object.keys(data).forEach((k:string) => {

 429        if (data[k] !== null && data[k] !== undefined) { newObject[k] = data[k]; }

 430    });

 431    return newObject;

 432}

 434/**

 435 * Compose authorization header

 436 * @param params 

 437 */

 438function compileParams(params:any) {

 439    let parts = [];

 440    for (let i in params) {

 441        if (typeof params[i] !== 'function') {

 442            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 443            parts.push(param);

 444        }

 445    }

 446    return 'Digest ' + parts.join(',');

 447}

 458    digestHeader = digestHeader.split('Digest ')[1];

 459    const params = {};

 460    digestHeader.split(',').forEach((part:string) => { 

 461        const kv = part.split('=').map((v:string) => v.trim());

 462        params[kv[0]] = kv[1].replace(/\"/g, '');

 464    return params;
