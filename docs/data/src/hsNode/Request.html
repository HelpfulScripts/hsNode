Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET and POST. The module builds upon 

   3 * [hsUtils.Request](https://helpfulscripts.github.io/hsUtil/#!/api/hsUtil/hsUtil.Request)

   4 * and provides support for

   5 * - simple authentication methods, 

   6 * - local caching,  

   7 * - decoding of received comntent.

   8 * - request pacing

   9 * 

  10 * ### Authentication

  11 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

  12 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  13 * The following authentication schemes are currently supported:

  14 * - Basic

  15 * - Digest

  16 * 

  17 * ### Caching

  18 * If caching is enabled, `GET` requests will attempt to return an available cached content

  19 * before issuing the request to the server.

  20 * Caching is disabled by default and can be enabled by setting a caching directory location before any `GET` call

  21 * `request.cache = './data'`.

  22 * Once set, individual `request.get(url)` requests will by default use caching. This can be 

  23 * disabled on a per call basis by providing `false` as a second parameter:

  24 * `request.get(url, false)` behaves as if `cache` is undefined.

  25 * Likewise, to disable all caching on future `GET` calls, set 

  26 * `request.cache = undefined`.

  27 * 

  28 * ### Decoding

  29 * Content received from a server or a cache can be decoded before returning the result. 

  30 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  31 * `request.decode = Request.decoders.str2json`

  32 * 

  33 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  34 * - str2json applies JSON.parse to the content to return an object literal

  35 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  36 * 

  37 * ### Pacing

  38 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  39 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  40 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  41 * 

  42 * ### Usage:

  43 * ```

  44 * import { Request } from 'hsnode';

  45 * const request = new Request.Request();

  46 * request.cache = __dirname + '/../_data/cache';

  47 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  48 * request.setAuthToken(apptokens.token);

  49 * const content = await request.get('http://...');

  50 * ```

  51 */

  86     * the location to use for caching. Set this property to the caching directory, e.g.:

  87     * `request.cache = './bin'`, ommitting a trailing `/`. 

  88     * To disable caching, set it to `undefined`.

  92    protected getURL(url:string|URL):URL { 

  93        return (typeof url === 'string')? new URL(url) : url; 

  94    }

  96    /**

  97     * constructs the cache name to use. The function call can be overwritten with 

  98     * a custom function to modify cache locations. 

  99     * This default implementation uses `request.cache/` as a prefix and adds 

 100     * the `path` element in `Options` to create required subdirectories 

 101     * underneath the `cache` location.

 102     * @param options the request options

 103     */

 104    public cacheName = (options:Options):string =>

 105        this.cache===undefined? undefined :  //   'q=.../' --> 'q=...-'    remove ?

 106            `${this.cache}/${options.path.replace(/q=(.*?)\//g,'q=$1-').replace(/\?/g,'')}`

 107

 108    /**

 109     * attempts to read a cached file and returns `undefined` if none is found.

 110     * The attempt consists of two steps:

 111     * - return a file named `<fname>.txt` as a text file, if available

 112     * - return a file names `<fname>.bin` as a binary file, if available

 113     * @param fname the path and name of the file, without extension

 114     * 

 115     */

 116    protected async readCached(fname:string):Promise<Response> {

 117        try {

 118            const meta = JSON.parse(await fs.readTextFile(`${fname}-meta.json`)); 

 119            const data = await fs.readFile(fname+'.bin', false);

 120            if (this.pace ) {

 121                this.log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) found cache for ${fname} `); 

 122            } else {

 123                this.log.transient(`found cache for ${fname} `); 

 124            }

 125            return {response:meta, data: data};   

 126        } catch(e) {

 127            return undefined; // no cache found

 128        }

 129    }

 130

 131    protected async writeCached(fname:string, response:Response) {

 132        try {

 133            // this.log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) received ${fname} `); 

 134            const type = response.response.headers["content-type"];

 135            const meta = {

 136                headers: {...response.response.headers},

 137                statusCode: response.response.statusCode,

 138                statusMessage: response.response.statusMessage,

 139                txt: response.response.txt

 140            }

 141            await fs.writeTextFile(`${fname}-meta.json`, JSON.stringify(meta));

 142            await fs.writeFile(`${fname}.bin`, <string>response.data, false);

 143       } catch(e) { this.log.warn(`error writing cache for content ${response.response.headers["content-type"]} and file ${fname}: ${e}`); }

 144    }

 146    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise<Response> {

 147        const fname = this.cacheName(options);

 148        if (fname && useCached && options.method === 'GET') { 

 149            const result = await this.readCached(fname); 

 150            if (result !== undefined) { return result; }

 151        }

 152

 153        const response = await super.requestOptions(options, useCached, postData);

 154        

 155        const code = response.response.statusCode||response.response.status;

 156        if(code >= 200 && code < 300) {

 157            if (fname && options.method === 'GET') {

 158                await this.writeCached(fname, response);

 159            }

 160        }

 161        return response;

 162    }

 166    protected async issueRequest(options:Options, postData?:any):Promise<Response> {

 167        const request = this;

 168        const prot = protocol[options.protocol];

 169        return new Promise((resolve:(out:Response)=>void, reject:(e:{data:string, error:any})=>void) => {

 170            let data = ''; 

 171            // this.log.info(()=>`requesting ${this.log.inspect(options, {depth:4})}`);

 172            const req = prot.request(options, (res:any) => {

 173                const encoding = request.isTextualContent(res.headers['content-type'])? 'utf8' : 'binary';

 174                res.setEncoding(encoding);

 175                res.txt = encoding === 'utf8';

 176                res.on('data', (chunk:string) => data += chunk);

 177                res.on('end', () => {

 178                    resolve({data:data, response:res});

 179                });

 180            });

 181            req.on('error', (e:any) => {

 182                reject({data:'', error:e});

 183            });

 184    

 185            // write data to request body

 186            if (postData !== undefined) { req.write(postData); }

 187            req.end();

 188        });

 189    }
