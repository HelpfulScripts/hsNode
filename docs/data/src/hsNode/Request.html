Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET and POST. The module builds upon 

   3 * [hsUtils.Request](https://helpfulscripts.github.io/hsUtil/#!/api/hsUtil/hsUtil.Request)

   4 * and provides support for

   5 * - simple authentication methods, 

   6 * - local caching,  

   7 * - decoding of received comntent.

   8 * - request pacing

   9 * 

  10 * ### Authentication

  11 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

  12 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  13 * The following authentication schemes are currently supported:

  14 * - Basic

  15 * - Digest

  16 * 

  17 * ### Caching

  18 * If caching is enabled, `GET` requests will attempt to return an available cached content

  19 * before issuing the request to the server.

  20 * Caching is disabled by default and can be enabled by setting a caching directory location before any `GET` call

  21 * `request.cache = './data'`.

  22 * Once set, individual `request.get(url)` requests will by default use caching. This can be 

  23 * disabled on a per call basis by providing `false` as a second parameter:

  24 * `request.get(url, false)` behaves as if `cache` is undefined.

  25 * Likewise, to disable all caching on future `GET` calls, set 

  26 * `request.cache = undefined`.

  27 * 

  28 * ### Decoding

  29 * Content received from a server or a cache can be decoded before returning the result. 

  30 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  31 * `request.decode = Request.decoders.str2json`

  32 * 

  33 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  34 * - str2json applies JSON.parse to the content to return an object literal

  35 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  36 * 

  37 * ### Pacing

  38 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  39 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  40 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  41 * 

  42 * ### Usage:

  43 * ```

  44 * import { Request } from 'hsnode';

  45 * const request = new Request.Request();

  46 * request.cache = __dirname + '/../_data/cache';

  47 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  48 * request.setAuthToken(apptokens.token);

  49 * const content = await request.get('http://...');

  50 * ```

  51 */

  80    protected getURL(url:string|URL):URL { 

  81        return (typeof url === 'string')? new URL(url) : url; 

  82    }

  84    /**

  85     * attempts to read a cached file and returns `undefined` if none is found.

  86     * The attempt consists of two steps:

  87     * - return a file named `.txt` as a text file, if available

  88     * - return a file names `.bin` as a binary file, if available

  89     * @param fname the path and name of the file, without extension

  90     * 

  91     */

  92    protected async readCached(fname:string):Promise {

  93        try {

  94            const meta = JSON.parse(await fs.readTextFile(`${fname}-meta.json`)); 

  95            const data = await fs.readFile(fname+'.bin');

  96            if (this.pace ) {

  97                this.log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) found cache for ${fname} `); 

  98            } else {

  99                this.log.transient(`found cache for ${fname} `); 

 100            }

 101            return {response:meta, data: data};   

 102        } catch(e) {

 103            return undefined; // no cache found

 104        }

 105    }

 106

 107    protected async writeCached(fname:string, response:Response) {

 108        try {

 109            // this.log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) received ${fname} `); 

 110            const type = response.response.headers["content-type"];

 111            const meta = {

 112                headers: Object.assign({}, response.response.headers),

 113                statusCode: response.response.statusCode,

 114                statusMessage: response.response.statusMessage,

 115                txt: response.response.txt

 116            }

 117            await fs.writeTextFile(`${fname}-meta.json`, JSON.stringify(meta));

 118            await fs.writeFile(`${fname}.bin`, response.data, false);

 119       } catch(e) { this.log.warn(`error writing cache for content ${response.response.headers["content-type"]} and file ${fname}: ${e}`); }

 120    }

 122    protected async issueRequest(options:Options, postData?:any):Promise {

 123        const request = this;

 124        const prot = protocol[options.protocol];

 125        return new Promise((resolve:(out:Response)=>void, reject:(e:{data:string, error:any})=>void) => {

 126            let data = ''; 

 127            this.log.debug(()=>`requesting ${this.log.inspect(options, {depth:4})}`);

 128            const req = prot.request(options, (res:any) => {

 129                const encoding = request.isTextualContent(res.headers['content-type'])? 'utf8' : 'binary';

 130                res.setEncoding(encoding);

 131                res.txt = encoding === 'utf8';

 132                res.on('data', (chunk:string) => data += chunk);

 133                res.on('end', () => {

 134                    resolve({data:data, response:res});

 135                });

 136            });

 137            req.on('error', (e:any) => {

 138                reject({data:'', error:e});

 139            });

 140    

 141            // write data to request body

 142            if (postData !== undefined) { req.write(postData); }

 143            req.end();

 144        });

 145    }
