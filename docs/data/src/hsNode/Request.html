Request.ts

   1/**

   2 * Utility functions for HTTP and HTTPS GET and POST. The module wraps the respective `Node` calls in async/await patterns

   3 * and provides support for

   4 * - simple authentication methods, 

   5 * - local caching,  

   6 * - decoding of received comntent.

   7 * - request pacing

   8 * 

   9 * ### Authentication

  10 * Authentication is enabled by setting {@link Request.Request.setCredentials `request.setCredentials`} before any call to 

  11 * {@link Request.Request.get `request.get`} or {@link Request.Request.post `request.post`}.

  12 * The following authentication schemes are currently supported:

  13 * - Basic

  14 * - Digest

  15 * 

  16 * ### Caching

  17 * If caching is enabled, `GET` requests will attempt to return an available cached content

  18 * before issuing the request to the server.

  19 * Caching is disabled by default and can be enabled by setting a caching directory location before any `GET` call

  20 * `request.cache = './data'`.

  21 * Once set, individual `request.get(url)` requests will by default use caching. This can be 

  22 * disabled on a per call basis by providing `false` as a second parameter:

  23 * `request.get(url, false)` behaves as if `cache` is undefined

  24 * 

  25 * ### Decoding

  26 * Content received from a server or a cache can be decoded before returning the result. 

  27 * Decoding is disabled by default and can be enabled by setting a `Decoder` function:

  28 * `request.decode = Request.decoders.str2json`

  29 * 

  30 * For convenience, the following predefined decoder functions are provided as static members of `Request.decoders`:

  31 * - str2json applies JSON.parse to the content to return an object literal

  32 * - html2json applies (html2json)[https://github.com/jxck/html2json#readme] to the content

  33 * 

  34 * ### Pacing

  35 * Enable request pacing via `request.setPace(ms, max)`. When enabled, requests will be issued with a minimal 

  36 * amount of `ms` milliseconds between them, and there will never be more than `max` number of calls pending.

  37 * All additional calls will be placed in a queue until the `ms` abd `max` conditions allow calling them.

  38 * 

  39 * ### Usage:

  40 * ```

  41 * import { Request } from 'hsnode';

  42 * const request = new Request.Request();

  43 * request.cache = __dirname + '/../_data/cache';

  44 * request.decode = (options:http.Options) => options.path.indexOf('binData?id=') < 0;

  45 * request.setAuthToken(apptokens.token);

  46 * const content = await request.get('http://...');

  47 * ```

  48 */

 227    /**

 228     * issues a `GET` or `POST` call and returns the `url` response. 

 229     * If `usedCache` is `true`, the cached version will be returned if available. 

 230     * Otherwise an `http` or `https` request is initiated, the result cached and returned.

 231     * If `decode` is defined it is called with the retrieved data before returning the result. 

 232     * @param url the url to retrieve

 233     * @param method 'GET' or 'POST'

 234     * @param useCached if `true`, attempt to return a cahced version. If `false`, or no cached version

 235     * exists, a remote call will be attempted.

 236     * @param postData data to submit for a `POST` call

 237     */

 238    protected async decodedRequest(url:string|URL, method:'GET'|'POST', useCached:boolean, postData?:any):Promise {

 239        const options = this.getOptions(url, method);

 240        try { 

 241            const result = await this.requestOptions(options, useCached, postData); 

 242            return this.decode? this.decode(result, options) : result;

 243        }

 244        catch(e) { return undefined; }

 245    }

 247    /**

 248     * attempts to read a cached file and returns `undefined` if none is found.

 249     * The attempt consists of two steps:

 250     * - return a file named `.txt` as a text file, if available

 251     * - return a file names `.bin` as a binary file, if available

 252     * @param fname the path and name of the file, without extension

 253     * 

 254     */

 255    protected async readCached(fname:string) {

 256        let result;

 257        try {

 258            if (await fs.isFile(fname+'.txt')) {

 259                result = await fs.readTextFile(`${fname}.txt`); 

 260            } else if (await fs.isFile(fname+'.bin')) {

 261                result = await fs.readFile(`${fname}.bin`, false); 

 262            }

 263        } catch(e) {}

 264        if (result) { log.transient(`found cache for ${fname} `); }

 265        return result;   // no cache found

 266    }

 268    protected async writeCached(fname:string, data:any, contentType:string) {

 269        try {

 270            let txt = false;

 271            if (contentType===undefined) { contentType = 'text/html'; }

 272            else { contentType = contentType.split(';')[0]; }

 273            const subTypes = contentType.split('/');

 274            switch (subTypes[0]) {

 275                case 'text':        txt = true; break;

 276                case 'image':       

 277                case 'audio':

 278                case 'font':        break;

 279                case 'application': switch(subTypes[1]) {

 280                    case 'json':    txt = true; break;

 281                    case 'pdf':     break;

 282                    default: log.debug(`caching ${contentType} as binary for ${fname}`);    

 283                }

 284                break;

 285                default: log.warn(`caching ${contentType} as binary for ${fname}`);    

 286            }

 287            return await (txt? fs.writeTextFile(`${fname}.txt`, data) : fs.writeFile(`${fname}.bin`, data, false));

 288        } catch(e) { log.warn(`error writing cache for content ${contentType} and file ${fname}: ${e}`); }

 289    }

 291    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise {

 292        const fname = this.cache? this.cacheName(options) : undefined;

 293        if (fname && useCached && options.method === 'GET') { 

 294            const result = await this.readCached(fname); 

 295            if (result !== undefined) { return result; }

 296        }

 297

 298        let response: Response;

 299        const err = <{statusCode:string|number, statusMessage:string, url:string}>{};

 300        try { 

 301            if (this.pace) { 

 302                log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) requesting ${options.url}`); 

 303                response = await this.pace.add(() => this.request(options, postData));

 304                log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()})             `);

 305            }

 306            else { 

 307                log.info(`requesting ${options.url}`); 

 308                response = await this.request(options, postData);

 309            }

 310            if((response.response.statusCode||response.response.status) === 200) {

 311                if (fname && options.method === 'GET') {

 312                    await this.writeCached(fname, response.data, response.response.headers["content-type"]);

 313                }

 314                return response.data;

 315            }

 316            log.warn(`${response.response.statusCode||response.response.status} requesting ${options.url}: ${response.response.statusMessage}`);

 317            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 318            err.statusMessage = response.response.statusMessage || 'no message';

 319        }

 320        catch(e) {

 321            log.warn(`error requesting ${options.url}: ${e.error.code}\n${log.inspect(e, 2)}`);

 322            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 323            err.statusMessage = JSON.stringify(e);

 324        }

 325        err.url = options.url;  

 326        fs.writeTextFile(fname + '.error.txt', JSON.stringify(err)); 

 327        throw(`response code ${err.statusCode} for ${options.url}`);

 328    }

 330    protected async request(options:Options, postData?:any):Promise {

 331        const httpProt:string = options.protocol.slice(0,-1);

 332        const http = protocol[httpProt];

 333        let auth = (options.headers && options.headers.Authorization);

 334        const response = await new Promise((resolve:(out:Response)=>void, reject:(e:{data:string, error:any})=>void) => {

 335            let data = ''; 

 336            log.debug(`requesting ${log.inspect(options, 4)}`);

 337            const req = http.request(options, (res:any) => {

 338                const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';

 339                log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);

 340                res.setEncoding(encoding);

 341                res.on('data', (chunk:string) => data += chunk);

 342                res.on('end', () => resolve({data:data, response:res}));

 343            });

 344            req.on('error', (e:any) => reject({data:'', error:e}));

 345    

 346            // write data to request body

 347            if (postData !== undefined) { req.write(postData); }

 348            req.end();

 349        });

 350        const prot = response.response.headers['www-authenticate'];

 351        if (this.credentials && prot) { 

 352            if (prot.trim().indexOf('Digest') === 0) {

 353                const digest = new Digest(this.credentials.user, this.credentials.password);

 354                digest.testDigestAuth(options, response.data, response.response);

 355                return this.request(options);

 356            }

 357            if (prot.trim().indexOf('Basic') === 0) {

 358                options.headers.Authorization = 'Basic ' + _btoa(`${this.credentials.user}:${this.credentials.password}`);

 359                return this.request(options);

 360            }

 361        } else {

 362            return response; 

 363        } 

 364    }

 383class Digest {

 384    nc = 0;

 385    username:string;

 386    password:string;

 388    constructor(username:string, password:string) {

 389        this.username = username;

 390        this.password = password;

 393    /**

 394     * Update and zero pad nc

 395     */

 396    updateNC():string {

 397        let max = 99999999;

 398        let padding = new Array(8).join('0') + '';

 399        this.nc = (this.nc > max ? 1 : this.nc + 1);

 400        let nc = this.nc + '';

 401        return padding.substr(0, 8 - nc.length) + nc;

 402    }

 404    /**

 405     * Parse challenge digest

 406     * @param qop 

 407     */

 408    generateCNONCE(qop:string) {

 409        let cnonce:any;

 410        let nc:string;

 411

 412        if (typeof qop === 'string') {

 413            let cnonceHash = createHash('md5');

 414            cnonceHash.update(Math.random().toString(36));

 415            cnonce = cnonceHash.digest('hex').substr(0, 16);

 416            nc = this.updateNC();

 417        }

 418        return {cnonce: cnonce, nc: nc};

 419    }

 421    /**

 422     * 

 423     * @param options 

 424     * @param data 

 425     * @param response 

 426     */

 427    testDigestAuth(options:Options, data:string, response:IncomingMessage): Options {

 428        log.debug(`received www-authenticate request for ${options.host}`);

 429

 430        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

 431        let ha1 = createHash('md5');

 432        let _str = `${this.username}:${challenge.realm}:${this.password}`;

 433        ha1.update(_str);

 434        let ha2 = createHash('md5');

 435        _str = `${options.method}:${options.path}`;

 436        ha2.update(_str);

 437    

 438        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

 439        let hash = createHash('md5');

 440        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

 441        hash.update(_str);

 442    

 443        // Setup response parameters

 444        let authParams:any = {

 445            realm: challenge.realm,

 446            username: this.username,

 447            uri: options.path,

 448            qop: challenge.qop,

 449            nonce: challenge.nonce,

 450            algorithm: "MD5",

 451            response: hash.digest('hex')

 452        };

 453    

 454        authParams = omitNull(authParams);

 455    

 456        if (cnonce) {

 457            authParams.nc = nc;

 458            authParams.cnonce = cnonce;

 459        }

 460    

 461        options.headers.Authorization = compileParams(authParams);

 462        return options;

 463    }

 464}

 468function omitNull(data:any) {

 469    let newObject = {};

 470    Object.keys(data).forEach((k:string) => {

 471        if (data[k] !== null && data[k] !== undefined) { newObject[k] = data[k]; }

 472    });

 473    return newObject;

 474}

 476/**

 477 * Compose authorization header

 478 * @param params 

 479 */

 480function compileParams(params:any) {

 481    let parts = [];

 482    for (let i in params) {

 483        if (typeof params[i] !== 'function') {

 484            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 485            parts.push(param);

 486        }

 487    }

 488    return 'Digest ' + parts.join(',');

 489}

 491/**

 492 * return `true` if double quotes are needed for `entry`

 493 * @param entry

 494 */

 495function putDoubleQuotes(entry:string) {

 496    return ['qop', 'nc'].indexOf(entry)<0;

 497}

 499function parseDigestResponse(digestHeader:string) {

 500    digestHeader = digestHeader.split('Digest ')[1];

 501    const params = {};

 502    digestHeader.split(',').forEach((part:string) => { 

 503        const kv = part.split('=').map((v:string) => v.trim());

 504        params[kv[0]] = kv[1].replace(/\"/g, '');

 505    });

 506    return params;

 507}

 509function isBinary(contentType:string) {

 510    const binary = {

 511        'text/html':    false,

 512        'text/plain':   false,

 513        'image/jpeg':   true,

 514        'image/png':    true

 515    };

 516    const result = binary[contentType];

 517    return (result === undefined)? false : result;

 518}
