  56/**

  57* Exension of [hsUtil.Log](https://helpfulscripts.github.io/hsUtil/#!/api/hsUtil/hsUtil.log), adding

  58* - support for {@link log.Log.logFile logging to a file}

  59* - support for {@link log.Log.transient transient } logging level

  60* - adaptation of `color` codes for use in a terminal

  61*/

  62export class Log extends LogUtil {

  63    public static log = new Log('');

  64

  65    /** name of the current log file, or undefined */

  66    protected LogFile: string; // initially disabled

  67

  68    constructor(prefix:string) { super(prefix); }

  69

  70    /**

  71     * reports a transient message to the log. This is an `INFO` level message that omits the

  72     * `line feed` character (i.e. `carriage return` only) so that it will be overwritten 

  73     * by the next logging output. 

  74     * @param msg the message to report. If msg is an object literal, a deep inspection will be printed.

  75     * @param log optional flag to enable/suppress logging to file. Defaults to `true`

  76     * @return promise to return the file written to, or undefined

  77     */

  78    public transient(msg:any):string { 

  79        return this.out(LogUtil.INFO, msg.padEnd(this.messageLength || 80, ' ')+' \r', { color: ['green']}); 

  80    }

  81

  82    /** 

  83     * the actual logging; overrides `output` in `hsUtil.Log`

  84     */

  85    protected output(color:string[], header:string, line:string) {

  86        const c = color.map(c => COLOR[c]).join('');

  87        const msg = `${c}${header}${COLOR['clear']} ${line}`;

  88        if (this.LogFile) { appendFileSync(date(this.LogFile), `${header} ${line}\n`); }

  89        if (line.slice(-1)==='\r') { process.stdout.write(msg); }

  90        else { console.log(msg); }

  91    }

  92

  93    /**

  94     * sets a new logfile name template. Logfiles are created using this template 

  95     * at the time of each log entry call. If the file exists, the log entry will be appended.

  96     * This is a global setting that affects reporting in all modules.

  97     * Usage:

  98     * - `logFile()`: return the current logfile name

  99     * - `logFile(null)`: disable log file

 100     * - `logFile('')`: set default log file template `log-%YYYY-%MM-%DD.txt`

 101     * - `logFile('log%D/%M/%Y.log')`: set new log file template

 102     * @param filePattern a template to use for log file names. Options for calling:

 103     * @return promise to return the current logfile name, or `undefined` if loggimng is disabled.

 104     */

 105    public async logFile(filePattern?:string):Promise {

 106        if (filePattern === null) {                    // disable logging in file

 107            this.LogFile = undefined; 

 108            this.info("disabling logfile");

 109            return this.LogFile;

 110        } else if (filePattern === undefined) {        // leave this.LogFile unchanged, return promise for logfile name

 111            return this.LogFile===undefined? undefined : date(this.LogFile);

 112        } else if (filePattern.indexOf('/')>=0) { 

 113            const parts = path.parse(filePattern);

 114            try {

 115                const exists:boolean = await pathExists(parts.dir);

 116                if (!exists) {

 117                    this.LogFile = undefined;

 118                    this.warn(`path '${parts.dir}' doesn't exists; logfile disabled`);

 119                } else {

 120                    this.LogFile = filePattern;

 121                    this.info("now logging to file " + date(filePattern));

 122                }

 123                return this.LogFile;

 124            } catch(e) {

 125                this.LogFile = undefined; 

 126                this.error(`checking path ${parts.dir}; logfile disabled`);

 127                return this.LogFile;

 128            }

 129        } else if (filePattern === '') {

 130            this.LogFile = 'log-%YYYY-%MM-%DD.txt';

 131        } else {

 132            this.LogFile=filePattern;

 133        }

 134        this.info(this.LogFile? `now logging to file ${date(this.LogFile)}` : 'logfile disbaled');

 135        return this.LogFile;

 136    }

 138    protected getPrePostfix(indent:string, level:number, currIndent:string, colors:string[]):[string,string,string,string] {

 139        const color  = colors? COLOR[colors[level % colors.length]] : '';

 140        return [`${currIndent}${indent}${color}`, colors? COLOR.clear : '', '\n', currIndent];

 141    }

 142}
