  14/** shell color escape codes */

  15const color = {

  16    red:    '\x1b[31m',

  17    yellow: '\x1b[33m',

  18    blue:   '\x1b[36m',

  19    green:  '\x1b[32m',

  20    bold:   '\x1b[1m',

  21    clear:  '\x1b[0m'

  22};

  23

  24/** boolean determining if log will be printed in color */

  25let gColors = true;

  26

  27

  28export class LogServer extends LogUtil {

  29    public static log = new LogServer('');

  30

  31    /** name of the current log file, or undefined */

  32    protected LogFile: string; // initially disabled

  33

  34    /** temporary color setting, defined in call to `inspect()` and used in `getPrePostfix` */

  35    protected inspectColors: string[];

  36

  37    constructor(prefix:string) { super(prefix); }

  38

  39    /**

  40     * reports a transient message to the log. This is an `INFO` level message that omits the

  41     * `line feed` character (i.e. `carriage return` only) so that it will be overwritten 

  42     * by the next logging output. Internally 

  43     * @param msg the message to report. If msg is an object literal, a deep inspection will be printed.

  44     * @param log optional flag to enable/suppress logging to file. Defaults to `true`

  45     * @return promise to return the file written to, or undefined

  46     */

  47    public transient(msg:any):string { 

  48        return this.out(LogUtil.INFO, msg+'\r'); 

  49    }

  50

  51    /**

  52     * configures the log facility.

  53     * - cfg.colors: boolean, determines if output is colored

  54     * - cfg.logfile: sets the naming template for the logfile. Set logFile=null to disable.

  55     * - cfg.format: sets the format for the timestamp for each log entry

  56     * - cfg.level: sets the reporting level (same as calling log.level())

  57     * @param cfg 

  58     */

  59    public config(cfg:{colors?:boolean, format?:string, level?:string }) {

  60        super.config(cfg);

  61        if (cfg.colors!==undefined) { gColors = cfg.colors; }   // true / false

  62    }

  63

  64    /** 

  65     * Creates the core format of the reported message. This method is 

  66     * called within `out` to allow for 

  67     * - format extensions using color codes.

  68     * - logging messages to a file.

  69     */

  70    protected makeMessage(line:string, lvl:string, dateStr:string, desc:string):string {

  71        const colors = { [LogServer.ERROR]: color.red+color.bold, [LogServer.WARN]: color.yellow+color.bold, [LogServer.DEBUG]: color.blue, [LogServer.INFO]: color.green };

  72        const logLine = super.makeMessage(line, lvl, dateStr, desc);

  73        if (this.LogFile) {

  74            appendFile(date(this.LogFile), logLine+'\n');

  75        }

  76        const colorLine = `${colors[lvl]||''} ${dateStr} ${this.reportPrefix} ${desc} ${color.clear} ${line}`;

  77        return gColors? colorLine : logLine;

  78    }

  79

  80    /**

  81     * sets a new logfile name template. Logfiles are created using this template 

  82     * at the time of each log entry call. If the file exists, the log entry will be appended.

  83     * This is a global setting that affects reporting in all modules.

  84     * Usage:

  85     * - `logFile()`: return the current logfile name

  86     * - `logFile(null)`: disable log file

  87     * - `logFile('')`: set default log file template `log-%YYYY-%MM-%DD.txt`

  88     * - `logFile('log%D/%M/%Y.log')`: set new log file template

  89     * @param file a template to use for log file names. Options for calling:

  90     * @return promise to return the current logfile name, or `undefined` if loggimng is disabled.

  91     */

  92    public async logFile(file?:string):Promise {

  93        if (file === null) {                    // disable logging in file

  94            this.LogFile = undefined; 

  95            this.info("disabling logfile");

  96            return this.LogFile;

  97        } else if (file === undefined) {        // leave this.LogFile unchanged, return promise for logfile name

  98            return date(this.LogFile);

  99        } else if (file.indexOf('/')>=0) { 

 100            const parts = path.parse(file);

 101            try {

 102                const exists:boolean = await pathExists(parts.dir);

 103                if (!exists) {

 104                    this.LogFile = undefined;

 105                    this.warn(`path '${parts.dir}' doesn't exists; logfile disabled`);

 106                } else {

 107                    this.LogFile = file;

 108                    this.info("now logging to file " + date(file));

 109                }

 110                return this.LogFile;

 111            } catch(e) {

 112                this.LogFile = undefined; 

 113                this.error(`checking path ${parts.dir}; logfile disabled`);

 114                return this.LogFile;

 115            }

 116        } else if (file === '') {

 117            this.LogFile = 'log-%YYYY-%MM-%DD.txt';

 118        } else {

 119            this.LogFile=file;

 120        }

 121        this.info(this.LogFile? `now logging to file ${date(this.LogFile)}` : 'logfile disbaled');

 122        return this.LogFile;

 123    }

 124

 125    /**

 126     * returns a string representation of an object literal, similar to the Node `util.inspect` call.

 127     * 

 128     * Usage: `log.info(log.inspect(struct, 1))`

 129     * 

 130     * The call returns a raw formatted text string, or a HTM formatted string if `colors` is defined.

 131     * @param struct the object literal to inspect

 132     * @param depth depth of recursion, defaults to 1. Use `null` for infinite depth

 133     * @param indent the indent string to use, will accumulate for each level of indentation, defaults to ''

 134     * @param colors an array of css color values to apply to keywords in the inspected structure;

 135     * if present, `inspect` will generate HTML content instead of raw text, substituting any `space` characters

 136     * (' ') with ` `. The color applied to each keyword cycles through the array with each increasing level, 

 137     * and restarts at index 0 when the level exceeds the length of the array.

 138     */

 139    public inspect(msg:any, depth=3, indent='   ', colors:string[]=null):string {

 140        this.inspectColors = colors;

 141        if (colors) { indent = indent.replace(/ /g, ' '); }

 142        return super.inspect(msg, depth, indent, colors);

 143    }

 144

 145    protected getPrePostfix(indent:string, level:number, currIndent:string):[string,string] {

 146        let [prefix, postfix] = super.getPrePostfix(indent, level, currIndent);

 147        if (this.inspectColors) {

 148            const color  = this.inspectColors[level % this.inspectColors.length];

 149            prefix = `color:${color};'>${prefix}`;

 150            postfix = `${postfix}`;

 151        }

 152        return [prefix, postfix];

 153    }
