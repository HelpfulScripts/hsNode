   1/**

   2 * Logging convenience functions.

   3 * ## Usage

   4 * This module extends the [hsUtil.log](https://helpfulscripts.github.io/hsUtil/#!/api/hsUtil/hsUtil.log) logging module

   5 * by adding a log file hook.

   6 * 

   7 * ### Using a log file

   8 * ```

   9 * import { log as gLog } from 'hsnode';

  10 * const log = gLog('mymodule');  // optional, to use a module specific log; otherwise uses the global instance

  11 * log.format('%MM%DD');

  12 * log.info('by the way:'); // -> 0817 info by the way:

  13 * log.logFile('l%YY%MM');  // -> 0817 info now logging to file l1608.txt

  14 * log.logFile(null);       // -> 0817 disabling logfile

  15 * log.error('oh dear!');   // -> 0817 error *** oh dear!

  16 * ```

  17 * 

  18 */

  22export { LogType }              from 'hsutil';

  23import { log as uLog, LogType } from 'hsutil';

  24import { dirname, normalize}    from 'path';

  25import { pathExists }           from './fsUtil';

  26import { appendFile }           from './fsUtil';

  27

  28export const log:LogType = uLog('', node_logToFile, node_pathExists);

  29

  30async function node_pathExists(file:string):Promise {

  31    const dir = dirname(normalize(file));

  32    return await pathExists(dir);

  33}

  34

  35async function node_logToFile(filename:string, msg:string):Promise {

  36    return await appendFile(filename, msg+'\n');

  37}
