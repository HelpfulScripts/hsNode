   1/**
   2 * Logging convenience functions.
   3 * ## Usage
   4 * This module extends the [hsUtil.log](https://helpfulscripts.github.io/hsUtil/#!/api/hsUtil/hsUtil.log) logging module
   5 * by adding a log file hook.
   6 * 
   7 * ### Using a log file
   8 * 


   9 * log.format('%MM%DD');

  10 * log.info('by the way:'); // -> 0817 info by the way:

  11 * log.logFile('l%YY%MM');  // -> 0817 info now logging to file l1608.txt

  12 * log.logFile(null);       // -> 0817 disabling logfile

  13 * log.error('oh dear!');   // -> 0817 error *** oh dear!

  14 * 

  15 * 
  16 */

  18export { LogType }              from 'hsutil';
  19import { log as uLog, LogType } from 'hsutil';  export const log:LogType = uLog('', node_logToFile, node_pathExists);
  20import { dirname, normalize}    from 'path';
  21import { pathExists }           from './fsUtil';
  22import { appendFile }           from './fsUtil';

  26function node_pathExists(file:string):Promise {
  27    const dir = dirname(normalize(file));
  28    return pathExists(dir);
  29}

  31function node_logToFile(filename:string, msg:string):Promise {
  32    return appendFile(filename, msg+'\n')
  33    .catch(e => { 
  34        console.log(`error appending '${msg}' to file ${log.logFile()} | ${filename}: ${e}`); 
  35        return msg;
  36    });
  37}
