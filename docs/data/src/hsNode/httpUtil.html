  19/**

  20 * Decodes an xm or html string into a JSON representation

  21 * @param xml 

  22 */

  23export function xml2json(xml:string):any {

  24    let result:any;

  25    while(xml.length>0) {

  26        let tag:any = xml.match(/<.*?>/);

  27        if (tag && tag.length > 0) {

  28            tag = tag[0].substring(1, tag[0].length-1);     // strip '<' and '>'

  29            result = result || {};

  30            tag = getAttributes(tag, result);

  31            let start = xml.indexOf(`<${tag}`);

  32            let end  = xml.indexOf(``);

  33            if (end > 0) {

  34                const close = xml.indexOf('>', start);

  35                let content = xml.substring(close+1, end).trim();   // remove opening and closing tag

  36                result[tag] = this.xml2json(content);

  37                xml = xml.substring(end+tag.length+3).trim(); 

  38            } else {    // no closing tag

  39                //result[tag] = {};

  40                xml = xml.substring(xml.indexOf('>')+1).trim(); // remove tag

  41            }

  42        } else { // literal

  43            result = xml.slice(0);

  44            xml = '';

  45        }

  46    }

  47    return result;

  48}

  51/**

  52 * sends a http or https GET or POST request and promises to return the result.

  53 * @param url the URL to pass along to the GET or POST request

  54 * @param user an optional user {@link Digest Digest}

  55 * @param postData optional data to post. If provided, a POST request will be sent instead of the default GET 

  56 * @return promise to provide the result of the request.

  57 */

  58export function request(url:URL, user?:Digest, referer?:string, postData?:any):Promise {

  59    let options = {

  60        method:     postData? 'POST': 'GET',

  61        protocol:   url.protocol,

  62        host:       url.host,

  63        hostname:   url.hostname,

  64        port:       url.port,

  65        pathname:   url.pathname,

  66        path:       url.pathname + (url.search || ''),

  67        headers:    { 'User-Agent': 'helpful scripts' },

  68    };

  69    if (referer) { options.headers.referer = referer; }

  70    return requestOptions(options, user, postData);

  71}

  74/**

  75 * Establishes a caching of retrieved sites. The class uses a cache location provided

  76 * during construction. Each `get` call will return a cache for the site, if available.

  77 * Otherwise, a call to the site is initiated and the result is cached at the specified location.

  78 * Any returned error codes, such as 404 messages, are treated as valid responses and cached 

  79 * to be returned future in future calls.

  80 * 

  81 * ### Usage

  82 * ```

  83 * const cachedGet = new CachedHTTPRequest('./data/cache/');

  84 * const pageText = await cachedGet.request(url, '');

  85 * ```

  86 */

  87export class CachedHTTPRequest {

  88    /**

  89     * Construct a cached get at a specified location int he file system.

  90     * @param cacheLocation the location for the cache

  91     * @param user an optional {@link Digest `Digest`} authentication information

  92     */

  93    constructor(public cacheLocation: string, public user?:Digest) {

  94    }

  95

  96    //--------- private methods -------

  97    private async requestOnline(url:URL, fname:string) {

  98        const resp:any = await request(url, this.user);

  99        log.info(`requested ${url}`);

 100        await fs.writeTextFile(fname, resp.data);

 101        return resp.data;

 102    }

 103    

 104    private async requestOffline(fname:string) {

 105        log.info(`get cached '${fname}'`);

 106        return await fs.readTextFile(fname);

 107    }

 108

 109    //--------- public methods -------

 110

 111    /**

 112     * 

 113     * @param url the URL to request, supports GET and POST

 114     * @param useCached optional boolean: if `false`, a call to `request` will ignore 

 115     * any cached version of the response

 116     */

 117    public async request(url:URL, useCached=true) {

 118        const fname = `${this.cacheLocation}${url.host}/${url.pathname}${url.search}`;

 119        const exists = await fs.isFile(fname);

 120        return (exists && useCached)? 

 121            await this.requestOffline(fname) : await this.requestOnline(url, fname);

 122    }

 123}

 155/**

 156 * Implements a Digest authentication, used in {@link request `request`} call.

 157 */

 158export class Digest {

 159    nc = 0;

 160    username:string;

 161    password:string;

 162

 163    constructor(username:string, password:string) {

 164        this.username = username;

 165        this.password = password;

 166    }

 167

 168    /**

 169     * Update and zero pad nc

 170     */

 171    updateNC():string {

 172        let max = 99999999;

 173        let padding = new Array(8).join('0') + '';

 174        this.nc = (this.nc > max ? 1 : this.nc + 1);

 175        let nc = this.nc + '';

 176        return padding.substr(0, 8 - nc.length) + nc;

 177    }

 178

 179    /**

 180     * Parse challenge digest

 181     * @param qop 

 182     */

 183    generateCNONCE(qop:string) {

 184        let cnonce:any;

 185        let nc:string;

 186

 187        if (typeof qop === 'string') {

 188            let cnonceHash = createHash('md5');

 189            cnonceHash.update(Math.random().toString(36));

 190            cnonce = cnonceHash.digest('hex').substr(0, 16);

 191            nc = this.updateNC();

 192        }

 193        return {cnonce: cnonce, nc: nc};

 194    }

 195

 196    /**

 197     * 

 198     * @param options 

 199     * @param data 

 200     * @param response 

 201     */

 202    testDigestAuth(options:any, data:string, response:IncomingMessage): Promise {

 203        log.info(`received www-authenticate request for ${options.host}`);

 204

 205        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

 206        let ha1 = createHash('md5');

 207        let _str = `${this.username}:${challenge.realm}:${this.password}`;

 208        ha1.update(_str);

 209        let ha2 = createHash('md5');

 210        _str = `${options.method}:${options.path}`;

 211        ha2.update(_str);

 212    

 213        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

 214        let hash = createHash('md5');

 215        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

 216        hash.update(_str);

 217    

 218        // Setup response parameters

 219        let authParams:any = {

 220            realm: challenge.realm,

 221            username: this.username,

 222            uri: options.path,

 223            qop: challenge.qop,

 224            nonce: challenge.nonce,

 225            algorithm: "MD5",

 226            response: hash.digest('hex')

 227        };

 228    

 229        authParams = omitNull(authParams);

 230    

 231        if (cnonce) {

 232            authParams.nc = nc;

 233            authParams.cnonce = cnonce;

 234        }

 235    

 236        options.headers.Authorization = compileParams(authParams);

 237        return requestOptions(options);

 238    }

 255/**

 256 * Compose authorization header

 257 * @param params 

 258 */

 259function compileParams(params:any) {

 260    let parts = [];

 261    for (let i in params) {

 262        if (typeof params[i] !== 'function') {

 263            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 264            parts.push(param);

 265        }

 266    }

 267    return 'Digest ' + parts.join(',');

 268}

 278function parseDigestResponse(digestHeader:string) {

 279    digestHeader = digestHeader.split('Digest ')[1];

 280    const params = {};

 281    digestHeader.split(',').forEach((part:string) => { 

 282        const kv = part.split('=').map((v:string) => v.trim());

 283        params[kv[0]] = kv[1].replace(/\"/g, '');

 284    });

 285    return params;

 286}

 288function isBinary(contentType:string) {

 289    const binary = {

 290        'text/html':    false,

 291        'text/plain':   false,

 292        'image/jpeg':   true,

 293        'image/png':    true

 294    };

 295    const result = binary[contentType];

 296    return (result === undefined)? false : result;

 297}

 299function getAttributes(tag:string, result:any) {

 300    let fields = tag.split(/(?=([^"]*"[^"]*")*[^"]*$)\s+/g);

 301    tag = fields[0].trim();

 302    result[tag] = {};

 303    fields.map(f => {

 304        if (f) {

 305            let attrs = f.split('=');

 306            result[tag].attrs = result[tag].attrs || {};

 307            if (attrs.length>1) {

 308                result[tag].attrs[attrs[0].trim()] = attrs[1].replace(/\"/g, '').trim();

 309            }

 310        }

 311    });

 312    return tag;

 313}

 315function requestOptions(options:any, user?:Digest, postData?:any):Promise {

 316    const prot:any = {

 317        'http:': require('http'),

 318        'https:': require('https')

 319    };

 320    let auth = (options.headers && options.headers.Authorization);

 321    log.debug(`requesting ${log.inspect(options, 4)}`);

 322    return new Promise((resolve:(out:HttpResponse)=>void, reject:(e:string)=>void) => {

 323        let data = ''; 

 324        log.debug(`sending request ${auth? 'with authorization ':''}for ${options.protocol}//${options.host}:${options.port}${options.path}`); 

 325        const req = prot[options.protocol].request(options, (res:any) => {

 326            const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';

 327            log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);

 328            res.setEncoding(encoding);

 329            res.on('data', (chunk:string) => { data += chunk; });

 330            res.on('end', () => { log.debug(`received ${encoding}`); resolve({data:data, response:res}); });

 331        });

 332        req.on('error', (e:any) => reject(e));

 333

 334        // write data to request body

 335        if (postData !== undefined) { req.write(postData); }

 336        req.end();

 337    })

 338    .then((res:HttpResponse) => {

 339        if (user && res.response.headers['www-authenticate']) { 

 340            return user.testDigestAuth(options, res.data, res.response);

 341        } else {

 342            return res; 

 343        } 

 344    });
