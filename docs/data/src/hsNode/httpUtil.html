 164    protected async requestOptions(options:Options, useCached:boolean, postData?:any):Promise {

 165        const decode = this.decodeJSON(options);

 166        const pending = () => `  ${this.pending}`.slice(-this.pendingDigits);

 167        const fname = this.cacheBaseLocation? this.cacheName(options) : undefined;

 168        if (fname && useCached) { try { 

 169            return await this.readCached(fname, decode); 

 170        } catch(e) {}}

 171

 172        let response: Response;

 173        const err = <{statusCode:string, statusMessage:string, url:string}>{};

 174        try { 

 175            log.info(`(${this.pace.inQueue()} | ${this.pace.inProgress()}) requesting ${options.url}`);

 176            response = this.pace? await this.pace.add(() => this.request(options, postData)) : await this.request(options, postData);

 177            log.transient(`(${this.pace.inQueue()} | ${this.pace.inProgress()})             `);

 178            if((response.response.statusCode||response.response.status) === 200) {

 179                return this.writeCached(fname, decode, response.data);

 180            }

 181            log.warn(`${response.response.statusCode||response.response.status} requesting ${options.url}: ${response.response.statusMessage}`);

 182            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 183            err.statusMessage = response.response.statusMessage || 'no message';

 184        }

 185        catch(e) {

 186            log.warn(`error requesting ${options.url}: ${e.error.code}\n${log.inspect(e, 2)}`);

 187            err.statusCode = response.response.statusCode || response.response.status|| 'no code';

 188            err.statusMessage = JSON.stringify(e);

 189        }

 190        err.url = options.url;  

 191        fs.writeTextFile(fname + '.error.txt', JSON.stringify(err)); 

 192        throw(`response code ${err.statusCode} for ${options.url}`);

 193    }

 195    protected async request(options:Options, postData?:any):Promise {

 196        const httpProt:string = options.protocol.slice(0,-1);

 197        const http = protocol[httpProt];

 198        let auth = (options.headers && options.headers.Authorization);

 199        log.debug(`requesting ${log.inspect(options, 4)}`);

 200        const response = await new Promise((resolve:(out:Response)=>void, reject:(e:Response)=>void) => {

 201            let data = ''; 

 202            log.debug(`sending request ${auth? 'with authorization ':''}for ${options.protocol}//${options.host}:${options.port}${options.path}`); 

 203            const req = http.request(options, (res:any) => {

 204                const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';

 205                log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);

 206                res.setEncoding(encoding);

 207                res.on('data', (chunk:string) => data += chunk);

 208                res.on('end', () => resolve({data:data, response:res}));

 209            });

 210            req.on('error', (e:any) => reject({data:'', error:e}));

 211    

 212            // write data to request body

 213            if (postData !== undefined) { req.write(postData); }

 214            req.end();

 215        });

 216        if (this.user && response.response.headers['www-authenticate']) { 

 217            const digestOptions = this.user.testDigestAuth(options, response.data, response.response);

 218            return this.request(digestOptions);

 219        } else {

 220            return response; 

 221        } 

 222    }

 225/**

 226 * Decodes an xm or html string into a JSON representation

 227 * @param xml 

 228 */

 229export function xml2json(xml:string):any {

 230    let result:any;

 231    while(xml.length>0) {

 232        let tag:any = xml.match(/<.*?>/);

 233        if (tag && tag.length > 0) {

 234            tag = tag[0].substring(1, tag[0].length-1);     // strip '<' and '>'

 235            result = result || {};

 236            tag = getAttributes(tag, result);

 237            let start = xml.indexOf(`<${tag}`);

 238            let end  = xml.indexOf(``);

 239            if (end > 0) {

 240                const close = xml.indexOf('>', start);

 241                let content = xml.substring(close+1, end).trim();   // remove opening and closing tag

 242                result[tag] = this.xml2json(content);

 243                xml = xml.substring(end+tag.length+3).trim(); 

 244            } else {    // no closing tag

 245                //result[tag] = {};

 246                xml = xml.substring(xml.indexOf('>')+1).trim(); // remove tag

 247            }

 248        } else { // literal

 249            result = xml.slice(0);

 250            xml = '';

 251        }

 252    }

 253    return result;

 254}

 292     * Update and zero pad nc

 293     */

 294    updateNC():string {

 295        let max = 99999999;

 296        let padding = new Array(8).join('0') + '';

 297        this.nc = (this.nc > max ? 1 : this.nc + 1);

 298        let nc = this.nc + '';

 299        return padding.substr(0, 8 - nc.length) + nc;

 300    }

 302    /**

 303     * Parse challenge digest

 304     * @param qop 

 305     */

 306    generateCNONCE(qop:string) {

 307        let cnonce:any;

 308        let nc:string;

 309

 310        if (typeof qop === 'string') {

 311            let cnonceHash = createHash('md5');

 312            cnonceHash.update(Math.random().toString(36));

 313            cnonce = cnonceHash.digest('hex').substr(0, 16);

 314            nc = this.updateNC();

 315        }

 316        return {cnonce: cnonce, nc: nc};

 317    }

 319    /**

 320     * 

 321     * @param options 

 322     * @param data 

 323     * @param response 

 324     */

 325    testDigestAuth(options:Options, data:string, response:IncomingMessage): Options {

 326        log.debug(`received www-authenticate request for ${options.host}`);

 327

 328        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

 329        let ha1 = createHash('md5');

 330        let _str = `${this.username}:${challenge.realm}:${this.password}`;

 331        ha1.update(_str);

 332        let ha2 = createHash('md5');

 333        _str = `${options.method}:${options.path}`;

 334        ha2.update(_str);

 335    

 336        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

 337        let hash = createHash('md5');

 338        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

 339        hash.update(_str);

 340    

 341        // Setup response parameters

 342        let authParams:any = {

 343            realm: challenge.realm,

 344            username: this.username,

 345            uri: options.path,

 346            qop: challenge.qop,

 347            nonce: challenge.nonce,

 348            algorithm: "MD5",

 349            response: hash.digest('hex')

 350        };

 351    

 352        authParams = omitNull(authParams);

 353    

 354        if (cnonce) {

 355            authParams.nc = nc;

 356            authParams.cnonce = cnonce;

 357        }

 358    

 359        options.headers.Authorization = compileParams(authParams);

 360        return options;

 361    }

 378/**

 379 * Compose authorization header

 380 * @param params 

 381 */

 382function compileParams(params:any) {

 383    let parts = [];

 384    for (let i in params) {

 385        if (typeof params[i] !== 'function') {

 386            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 387            parts.push(param);

 388        }

 389    }

 390    return 'Digest ' + parts.join(',');

 391}

 401function parseDigestResponse(digestHeader:string) {

 402    digestHeader = digestHeader.split('Digest ')[1];

 403    const params = {};

 404    digestHeader.split(',').forEach((part:string) => { 

 405        const kv = part.split('=').map((v:string) => v.trim());

 406        params[kv[0]] = kv[1].replace(/\"/g, '');

 407    });

 408    return params;

 422function getAttributes(tag:string, result:any) {

 423    let fields = tag.split(/(?=([^"]*"[^"]*")*[^"]*$)\s+/g);

 424    tag = fields[0].trim();

 425    result[tag] = {};

 426    fields.map(f => {

 427        if (f) {

 428            let attrs = f.split('=');

 429            result[tag].attrs = result[tag].attrs || {};

 430            if (attrs.length>1) {

 431                result[tag].attrs[attrs[0].trim()] = attrs[1].replace(/\"/g, '').trim();

 432            }

 433        }

 434    });

 435    return tag;

 436}
