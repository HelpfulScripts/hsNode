  44    /**
  45     * Update and zero pad nc
  46     */
  47    updateNC():string {
  48        let max = 99999999;
  49        let padding = new Array(8).join('0') + '';
  50        this.nc = (this.nc > max ? 1 : this.nc + 1);
  51        let nc = this.nc + '';
  52        return padding.substr(0, 8 - nc.length) + nc;
  53    }

  55    /**
  56     * Parse challenge digest
  57     * @param qop 
  58     */
  59    generateCNONCE(qop:string) {
  60        let cnonce:any;
  61        let nc:string;
  62
  63        if (typeof qop === 'string') {
  64            let cnonceHash = createHash('md5');
  65            cnonceHash.update(Math.random().toString(36));
  66            cnonce = cnonceHash.digest('hex').substr(0, 16);
  67            nc = this.updateNC();
  68        }
  69        return {cnonce: cnonce, nc: nc};
  70    }

  72    /**
  73     * 
  74     * @param options 
  75     * @param data 
  76     * @param response 
  77     */
  78    testDigestAuth(options:any, data:string, response:IncomingMessage): Promise {
  79        log.info(`received www-authenticate request for ${options.host}`);
  80
  81        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);
  82        let ha1 = createHash('md5');
  83        let _str = `${this.username}:${challenge.realm}:${this.password}`;
  84        ha1.update(_str);
  85        let ha2 = createHash('md5');
  86        _str = `${options.method}:${options.path}`;
  87        ha2.update(_str);
  88    
  89        let {nc, cnonce} = this.generateCNONCE(challenge.qop);
  90        let hash = createHash('md5');
  91        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;
  92        hash.update(_str);
  93    
  94        // Setup response parameters
  95        let authParams:any = {
  96            realm: challenge.realm,
  97            username: this.username,
  98            uri: options.path,
  99            qop: challenge.qop,
 100            nonce: challenge.nonce,
 101            algorithm: "MD5",
 102            response: hash.digest('hex')
 103        };
 104    
 105        authParams = omitNull(authParams);
 106    
 107        if (cnonce) {
 108            authParams.nc = nc;
 109            authParams.cnonce = cnonce;
 110        }
 111    
 112        options.headers.Authorization = compileParams(authParams);
 113        return requestOptions(options);
 114    }

 133function compileParams(params:any) {
 134    let parts = [];
 135    for (let i in params) {
 136        if (typeof params[i] !== 'function') {
 137            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);
 138            parts.push(param);
 139        }
 140    }
 141    return 'Digest ' + parts.join(',');
 142}

 152function parseDigestResponse(digestHeader:string) {
 153    digestHeader = digestHeader.split('Digest ')[1];
 154    const params = {};
 155    digestHeader.split(',').forEach((part:string) => { 
 156        const kv = part.split('=').map((v:string) => v.trim());
 157        params[kv[0]] = kv[1].replace(/\"/g, '');
 158    });
 159    return params;

 173function getAttributes(tag:string, result:any) {
 174    let fields = tag.split(/(?=([^"]*"[^"]*")*[^"]*$)\s+/g);
 175    tag = fields[0].trim();
 176    result[tag] = {};
 177    if (fields.length>0) {
 178        for (let i=1; i  179            let attrs = fields[i].split('=');
 180            result[tag].attrs = result[tag].attrs || {};
 181            result[tag].attrs[attrs[0].trim()] = attrs[1].split('"')[1].trim();
 182        }
 183    }
 184    return tag;
 185}

 192/**
 193 * sends a http get request and promises to return the result.
 194 * @param url the URL to pass along to the GET or POST request
 195 * @param user an optional user {@link Digest Digest}
 196 * @param postData optional data to post. If provided, a POST request will be sent instead of the default GET 
 197 * @return promise to provide the result of the request.
 198 */
 199export function request(url:URL, user?:Digest, referer?:string, postData?:any):Promise {
 200    let options = {
 201        method:     postData? 'POST': 'GET',
 202        protocol:   url.protocol,
 203        host:       url.host,
 204        hostname:   url.hostname,
 205        port:       url.port,
 206        pathname:   url.pathname,
 207        path:       url.pathname + (url.search || ''),
 208        headers:    { 'User-Agent': 'helpful scripts' },
 209    };
 210    if (referer) { options.headers.referer = referer; }
 211    return requestOptions(options, user, postData);
 212}

 214function requestOptions(options:any, user?:Digest, postData?:any):Promise {
 215    let auth = (options.headers && options.headers.Authorization);
 216    log.debug(`requesting ${log.inspect(options, 4)}`);
 217    return new Promise((resolve:(out:HttpResponse)=>void, reject:(e:string)=>void) => {
 218        let data = ''; 
 219        log.debug(`sending request ${auth? 'with authorization ':''}for ${options.protocol}//${options.host}:${options.port}${options.path}`); 
 220        const req = http.request(options, (res:any) => {
 221            const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';
 222            log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);
 223            res.setEncoding(encoding);
 224            res.on('data', (chunk:string) => { data += chunk; });
 225            res.on('end', () => { log.debug(`received ${encoding}`); resolve({data:data, response:res}); });
 226        });
 227        req.on('error', (e:any) => reject(e));
 228
 229        // write data to request body
 230        if (postData !== undefined) { req.write(postData); }
 231        req.end();
 232    })
 233    .then((res:HttpResponse) => {
 234        if (user && res.response.headers['www-authenticate']) { 
 235            return user.testDigestAuth(options, res.data, res.response);
 236        } else {
 237            return res; 
 238        } 
 239    });
 240}

 244/**
 245 * Decodes an xm or html string into a JSON representation
 246 * @param xml 
 247 */
 248export function xml2json(xml:string):any {
 249    let result:any;
 250    while(xml.length>0) {
 251        let tag:any = xml.match(/<.*?>/);
 252        if (tag && tag.length > 0) {
 253            tag = tag[0].substring(1, tag[0].length-1);     // strip '<' and '>'
 254            result = result || {};
 255            tag = getAttributes(tag, result);
 256            let start = xml.indexOf(`<${tag}`);
 257            let end  = xml.indexOf(``);
 258            if (end > 0) {
 259                let content = xml.substring(start+tag.length+2, end).trim();   // remove opening and closing tag
 260                result[tag] = this.xml2json(content);
 261                xml = xml.substring(end+tag.length+3).trim(); 
 262            } else {    // no closing tag
 263                //result[tag] = {};
 264                xml = xml.substring(xml.indexOf('>')+1).trim(); // remove tag
 265            }
 266        } else { // literal
 267            result = xml.slice(0);
 268            xml = '';
 269        }
 270    }
 271    return result;
