  45    /**

  46     * Update and zero pad nc

  47     */

  48    updateNC():string {

  49        let max = 99999999;

  50        let padding = new Array(8).join('0') + '';

  51        this.nc = (this.nc > max ? 1 : this.nc + 1);

  52        let nc = this.nc + '';

  53        return padding.substr(0, 8 - nc.length) + nc;

  54    }

  56    /**

  57     * Parse challenge digest

  58     * @param qop 

  59     */

  60    generateCNONCE(qop:string) {

  61        let cnonce:any;

  62        let nc:string;

  63

  64        if (typeof qop === 'string') {

  65            let cnonceHash = createHash('md5');

  66            cnonceHash.update(Math.random().toString(36));

  67            cnonce = cnonceHash.digest('hex').substr(0, 16);

  68            nc = this.updateNC();

  69        }

  70        return {cnonce: cnonce, nc: nc};

  71    }

  73    /**

  74     * 

  75     * @param options 

  76     * @param data 

  77     * @param response 

  78     */

  79    testDigestAuth(options:any, data:string, response:IncomingMessage): Promise {

  80        log.info(`received www-authenticate request for ${options.host}`);

  81

  82        let challenge:any = parseDigestResponse(response.headers['www-authenticate']);

  83        let ha1 = createHash('md5');

  84        let _str = `${this.username}:${challenge.realm}:${this.password}`;

  85        ha1.update(_str);

  86        let ha2 = createHash('md5');

  87        _str = `${options.method}:${options.path}`;

  88        ha2.update(_str);

  89    

  90        let {nc, cnonce} = this.generateCNONCE(challenge.qop);

  91        let hash = createHash('md5');

  92        _str = `${ha1.digest('hex')}:${challenge.nonce}:${nc}:${cnonce}:${challenge.qop}:${ha2.digest('hex')}`;

  93        hash.update(_str);

  94    

  95        // Setup response parameters

  96        let authParams:any = {

  97            realm: challenge.realm,

  98            username: this.username,

  99            uri: options.path,

 100            qop: challenge.qop,

 101            nonce: challenge.nonce,

 102            algorithm: "MD5",

 103            response: hash.digest('hex')

 104        };

 105    

 106        authParams = omitNull(authParams);

 107    

 108        if (cnonce) {

 109            authParams.nc = nc;

 110            authParams.cnonce = cnonce;

 111        }

 112    

 113        options.headers.Authorization = compileParams(authParams);

 114        return requestOptions(options);

 115    }

 130/**

 131 * Compose authorization header

 132 * @param params 

 133 */

 134function compileParams(params:any) {

 135    let parts = [];

 136    for (let i in params) {

 137        if (typeof params[i] !== 'function') {

 138            let param = i + '=' + (putDoubleQuotes(i) ? `"${params[i]}"` : params[i]);

 139            parts.push(param);

 140        }

 141    }

 142    return 'Digest ' + parts.join(',');

 143}

 153function parseDigestResponse(digestHeader:string) {

 154    digestHeader = digestHeader.split('Digest ')[1];

 155    const params = {};

 156    digestHeader.split(',').forEach((part:string) => { 

 157        const kv = part.split('=').map((v:string) => v.trim());

 158        params[kv[0]] = kv[1].replace(/\"/g, '');

 159    });

 160    return params;

 161}

 174function getAttributes(tag:string, result:any) {

 175    let fields = tag.split(/(?=([^"]*"[^"]*")*[^"]*$)\s+/g);

 176    tag = fields[0].trim();

 177    result[tag] = {};

 178    fields.map(f => {

 179        if (f) {

 180            let attrs = f.split('=');

 181            result[tag].attrs = result[tag].attrs || {};

 182            if (attrs.length>1) {

 183                result[tag].attrs[attrs[0].trim()] = attrs[1].replace(/\"/g, '').trim();

 184            }

 185        }

 186    });

 187    return tag;

 188}

 195/**

 196 * sends a http or https get request and promises to return the result.

 197 * @param url the URL to pass along to the GET or POST request

 198 * @param user an optional user {@link Digest Digest}

 199 * @param postData optional data to post. If provided, a POST request will be sent instead of the default GET 

 200 * @return promise to provide the result of the request.

 201 */

 202export function request(url:URL, user?:Digest, referer?:string, postData?:any):Promise {

 203    let options = {

 204        method:     postData? 'POST': 'GET',

 205        protocol:   url.protocol,

 206        host:       url.host,

 207        hostname:   url.hostname,

 208        port:       url.port,

 209        pathname:   url.pathname,

 210        path:       url.pathname + (url.search || ''),

 211        headers:    { 'User-Agent': 'helpful scripts' },

 212    };

 213    if (referer) { options.headers.referer = referer; }

 214    return requestOptions(options, user, postData);

 215}

 217function requestOptions(options:any, user?:Digest, postData?:any):Promise {

 218    const prot:any = {

 219        'http:': require('http'),

 220        'https:': require('https')

 221    };

 222    let auth = (options.headers && options.headers.Authorization);

 223    log.debug(`requesting ${log.inspect(options, 4)}`);

 224    return new Promise((resolve:(out:HttpResponse)=>void, reject:(e:string)=>void) => {

 225        let data = ''; 

 226        log.debug(`sending request ${auth? 'with authorization ':''}for ${options.protocol}//${options.host}:${options.port}${options.path}`); 

 227        const req = prot[options.protocol].request(options, (res:any) => {

 228            const encoding = isBinary(res.headers['content-type'])? 'binary' : 'utf8';

 229            log.debug(`receiving...${res.headers['content-type']} => ${encoding}`);

 230            res.setEncoding(encoding);

 231            res.on('data', (chunk:string) => { data += chunk; });

 232            res.on('end', () => { log.debug(`received ${encoding}`); resolve({data:data, response:res}); });

 233        });

 234        req.on('error', (e:any) => reject(e));

 235

 236        // write data to request body

 237        if (postData !== undefined) { req.write(postData); }

 238        req.end();

 239    })

 240    .then((res:HttpResponse) => {

 241        if (user && res.response.headers['www-authenticate']) { 

 242            return user.testDigestAuth(options, res.data, res.response);

 243        } else {

 244            return res; 

 245        } 

 246    });

 247}

 251/**

 252 * Decodes an xm or html string into a JSON representation

 253 * @param xml 

 254 */

 255export function xml2json(xml:string):any {

 256    let result:any;

 257    while(xml.length>0) {

 258        let tag:any = xml.match(/<.*?>/);

 259        if (tag && tag.length > 0) {

 260            tag = tag[0].substring(1, tag[0].length-1);     // strip '<' and '>'

 261            result = result || {};

 262            tag = getAttributes(tag, result);

 263            let start = xml.indexOf(`<${tag}`);

 264            let end  = xml.indexOf(``);

 265            if (end > 0) {

 266                const close = xml.indexOf('>', start);

 267                let content = xml.substring(close+1, end).trim();   // remove opening and closing tag

 268                result[tag] = this.xml2json(content);

 269                xml = xml.substring(end+tag.length+3).trim(); 

 270            } else {    // no closing tag

 271                //result[tag] = {};

 272                xml = xml.substring(xml.indexOf('>')+1).trim(); // remove tag

 273            }

 274        } else { // literal

 275            result = xml.slice(0);

 276            xml = '';

 277        }

 278    }

 279    return result;
