  60/**

  61 * returns a promise for the stats of the file addressed by `thePath`.

  62 * If `thePath` addresses a symbolic link, the stats of the linked file are returned.

  63 * @param thePath 

  64 */

  65async function stat(thePath:string):Promise {

  66    const p = await realPath(thePath);

  67    return await new Promise((resolve, reject) => {

  68        fs.stat(p, (err:any, stats:fs.Stats) => {

  69            if(err) { 

  70                console.log(`error getting stats for ${thePath}: ${err}`);

  71                reject(err); 

  72            } // reject is hard to test: realpath throws an error before stat can.

  73            else { resolve(stats); }

  74        });

  75    });

  76}

  78/**

  79 * returns a promise for the stats of the file addressed by `thePath`.

  80 * If `thePath` addresses a symbolic link, the stats of the link are returned.

  81 * @param thePath 

  82 */

  83async function lstat(thePath:string):Promise {

  84    const p = path.normalize(thePath);

  85 return new Promise((resolve, reject) => 

  86        fs.lstat(p, (err:any, stats:any) => err? reject(err) : resolve(stats))

  87 );

  88}

  98/**

  99 * determines the canonical path for `thePath`, resolving all symbolic links and '../'in the path.

 100 * @param thePath the path to check

 101 * @return promise to provide the real canonical system path.

 102 */

 103export function realPath(thePath:string):Promise {

 104 return new Promise((resolve:(path:string)=>void, reject:(err:any)=>void) => {

 105 fs.realpath(thePath, (err:any, resolvedPath:string) => err? reject(err) : resolve(resolvedPath) );

 106    })

 107    .catch(error);

 108}

 163/**

 164 * creates any missing directories in `thePath` and promises to return the path name.

 165 * Characters after the last `/` in `thePath` will be interpreted as a filename, hence no directory willbe created form them.

 166 * Terminate `thePath` with a final `/` to indicate that all parts should be created.

 167 * FInally, for precaution `mkdirs` only creates directories within the current working directory.

 168 * @param thePath the path to check

 169 * @return promise to provide the path name

 170 */ 

 171export async function mkdirs(thePath:string):Promise {    

 172    const p = path.normalize(path.resolve(process.cwd(),thePath));

 173    // console.log(`mkdirs: '${thePath}' -> '${p}'`);

 174    let dirs = p.split('/');

 175    // create complete successive subdirs from the split

 176    dirs = dirs.map((dir, i) => dirs.slice(0,i+1).join('/'));

 177    for (let i=0; i

 178        const dir = dirs[i];

 179        const exists = await isDirectory(dir);

 180        if (!exists) { try {

 181            await fs.promises.mkdir(dir);

 182        } catch(e) { 

 183            if (e.code !== 'EEXIST') {

 184                console.error(`error in mkdirs: ${e.code}`);

 185                console.error(e); 

 186                throw `mkdir failed for ${dir}: ${p}\n${e}`;

 187            }

 188        }}

 189    }

 190    return p;

 191}

 193/**

 194 * lists all files in a directory and promises to provide the list.

 195 * @param thePath the path to check

 196 * @return promise to provide a list of directory entries.

 197 */

 198export async function readDir(thePath:string):Promise {

 199    const p = await realPath(thePath);

 200    return await new Promise((resolve:(files:any)=>void, reject:(err:any)=>void) => {

 201        fs.readdir(p, (err:any, files:any) =>  {

 202            if(err) { reject(err); }

 203            else { 

 204                files.path = p;

 205                resolve(files); 

 206            }

 207        });

 208    })

 209    .catch(error);

 210}

 250/**

 251 * writes a file either as binary or text and promises to return the file name.

 252 * @param thePath the path to write to

 253 * @param content the content to write

 254 * @param isText `true`|`false` if file should be read as `utf8`|binary 

 255 * @return promise to provide the file name if successful.

 256 */

 257export async function writeFile(thePath:string, content:string, isText:boolean=true):Promise {

 258    var encoding:any = isText? 'utf8' : 'binary';

 259    await mkdirs(path.dirname(thePath));

 260    return await new Promise((resolve, reject) => {

 261        fs.writeFile(thePath, content, encoding, (err:any) =>

 262            err? reject(`mkdirs failed in writeFile for '${thePath}': ${err}`) : resolve(thePath));

 263    }); 

 264}

 266/**

 267 * writes content to a file as a stream and promises to return the file name.

 268 * @param thePath the path to write to

 269 * @param content the content to write

 270 * @return promise to provide the file name if successful.

 271 */

 272export async function writeStream(thePath:string, content:string):Promise {

 273    return await new Promise((resolve, reject) => {

 274        const s = fs.createWriteStream(thePath, {flags:'w', mode:0o666});

 275        s.on('error', (src:any) => reject(`writeStream error '${src}' for path '${thePath}'`));

 276        s.write(content, 'binary', () => resolve(thePath));

 277        s.end();

 278    });

 279}

 306 * @param thePath the path to write to

 315    })

 320/**

 321 * promises to delete a file or folder and return the file or folder name.

 322 * @param thePath the path to write

 323 * @return promise to provide the name of the removed file.

 324 */

 325export async function remove(thePath:string):Promise {

 326    const dir:boolean = await isDirectory(thePath);

 327 return await new Promise((resolve, reject) => {

 328        dir? fs.rmdir(thePath, (e:any) => (e? reject(e) : resolve(thePath)))

 329           : fs.unlink(thePath, (e:any) => (e? reject(e) : resolve(thePath)));

 330 });

 331}

 333/**

 334 * promises to delete a file or folder and return the file or folder name.

 335 * @param thePath the path to write

 336 * @return promise to provide the list of names of the removed files.

 337 */

 338export async function removeAll(thePath:string):Promise {

 339    const removed = [];

 340    const dir:boolean = await isDirectory(thePath);

 341    if (dir) {

 342        const list = await readDir(thePath);

 343        await Promise.all(list.map(async i => removed.push(...await removeAll(`${thePath}/${i}`))));

 344    }

 345    removed.push(await remove(thePath));

 346    return removed;

 347}
