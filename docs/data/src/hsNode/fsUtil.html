  43/**

  44 * returns a promise for the stats of the file addressed by `thePath`.

  45 * If `thePath` addresses a symbolic link, the stats of the linked file are returned.

  46 * @param thePath 

  47 */

  48async function stat(thePath:string):Promise<fs.Stats> {

  49    const p = await realPath(thePath);

  50    return await new Promise((resolve, reject) => {

  51        fs.stat(p, (err:any, stats:fs.Stats) => {

  52            if(err) { 

  53                console.log(`error getting stats for ${thePath}: ${err}`);

  54                reject(err); 

  55            } // reject is hard to test: realpath throws an error before stat can.

  56            else { resolve(stats); }

  57        });

  58    });

  59}

  61/**

  62 * returns a promise for the stats of the file addressed by `thePath`.

  63 * If `thePath` addresses a symbolic link, the stats of the link are returned.

  64 * @param thePath 

  65 */

  66async function lstat(thePath:string):Promise<fs.Stats> {

  67    const p = path.normalize(thePath);

  68 return new Promise((resolve, reject) => 

  69        fs.lstat(p, (err:any, stats:any) => err? reject(err) : resolve(stats))

  70 );

  71}

  73function error(err:any):any {

  74    const msg = `*** error in fsUtil: ${err}, stack:\n${err.stack}`;

  75    throw new Error(msg);

  76}

  81/**

  82 * determines the canonical path for `thePath`, resolving all symbolic links and '../'in the path.

  83 * @param thePath the path to check

  84 * @return promise to provide the real canonical system path.

  85 */

  86export function realPath(thePath:string):Promise<string> {

  87 return new Promise((resolve:(path:string)=>void, reject:(err:any)=>void) => {

  88 fs.realpath(thePath, (err:any, resolvedPath:string) => err? reject(err) : resolve(resolvedPath) );

  89    })

  90    .catch(error);

  91}

 159/**

 160 * creates any missing directories in `thePath` and promises to return the path name.

 161 * Characters after the last `/` in `thePath` will be interpreted as a filename, hence no directory willbe created form them.

 162 * Terminate `thePath` with a final `/` to indicate that all parts should be created.

 163 * FInally, for precaution `mkdirs` only creates directories within the current working directory.

 164 * @param thePath the path to check

 165 * @return promise to provide the path name

 166 */ 

 167export async function mkdirs(thePath:string):Promise<string> {    

 168    const p = path.normalize(path.resolve(process.cwd(),thePath));

 169    // console.log(`mkdirs: '${thePath}' -> '${p}'`);

 170    let dirs = p.split('/');

 171    // create complete successive subdirs from the split

 172    dirs = dirs.map((dir, i) => dirs.slice(0,i+1).join('/'));

 173    for (let i=0; i<dirs.length; i++) {

 174        const dir = dirs[i];

 175        const exists = await isDirectory(dir);

 176        if (!exists) { try {

 177            await fs.promises.mkdir(dir);

 178        } catch(e) { 

 179            if (e.code !== 'EEXIST') {

 180                console.error(`error in mkdirs: ${e.code}`);

 181                console.error(e); 

 182                throw `mkdir failed for ${dir}: ${p}\n${e}`;

 183            }

 184        }}

 185    }

 186    return p;

 187}

 189/**

 190 * lists all files in a directory and promises to provide the list.

 191 * @param thePath the path to check

 192 * @return promise to provide a list of directory entries.

 193 */

 194export async function readDir(thePath:string):Promise<string[]> {

 195    const p = await realPath(thePath);

 196    return await new Promise((resolve:(files:any)=>void, reject:(err:any)=>void) => {

 197        fs.readdir(p, (err:any, files:any) =>  {

 198            if(err) { reject(err); }

 199            else { 

 200                files.path = p;

 201                resolve(files); 

 202            }

 203        });

 204    })

 205    .catch(error);

 206}

 209/**

 210 * reads a file either as binary or text and promises to provide the content.

 211 * @param thePath the path to read

 212 * @param isText [default=true] `true`|`false` if file should be read as `utf8`|binary 

 213 * @return promise to provide file content.

 215export function readFile(thePath:string, isText=true):Promise<any> {

 216 return new Promise((resolve:(data:any)=>void, reject:(err:any)=>void) => {

 217 fs.readFile(thePath, isText? 'utf8' : 'binary', (err:any, data:any) => 

 218            err? reject(err) : resolve(data));

 219 })

 220    .catch(error);

 246/**

 247 * writes a file either as binary or text and promises to return the file name.

 248 * @param thePath the path to write to

 249 * @param content the content to write

 250 * @param isText `true`|`false` if file should be read as `utf8`|binary 

 251 * @return promise to provide the file name if successful.

 252 */

 253export async function writeFile(thePath:string, content:string, isText:boolean=true):Promise<string> {

 254    var encoding:any = isText? 'utf8' : 'binary';

 255    await mkdirs(path.dirname(thePath));

 256    return await new Promise((resolve, reject) => {

 257        fs.writeFile(thePath, content, encoding, (err:any) =>

 258            err? reject(`mkdirs failed in writeFile for '${thePath}': ${err}`) : resolve(thePath));

 259    }); 

 260}

 262/**

 263 * writes content to a file as a stream and promises to return the file name.

 264 * @param thePath the path to write to

 265 * @param content the content to write

 266 * @return promise to provide the file name if successful.

 267 */

 268export async function writeStream(thePath:string, content:string):Promise<string> {

 269    return await new Promise((resolve, reject) => {

 270        const s = fs.createWriteStream(thePath, {flags:'w', mode:0o666});

 271        s.on('error', (src:any) => reject(`writeStream error '${src}' for path '${thePath}'`));

 272        s.write(content, 'binary', () => resolve(thePath));

 273        s.end();

 274    });

 275}

 301 * appends to a file either as binary or text and promises to return the file name.

 302 * @param thePath the path to write to

 303 * @param content the content to write

 304 * @param isText `true`|`false` if file should be read as `utf8`|binary 

 305 * @return promise to provide the realPath of the file written to.

 307export async function appendFile(thePath:string, content:string, isText:boolean=true):Promise<string> {

 308    var encoding:any = isText? 'utf8' : {encoding: null};

 309    try { return await new Promise((resolve, reject) => {

 310        fs.appendFile(thePath, content, encoding, (err:any) => err? reject(err) : resolve(thePath));

 311    })} catch(e) { error(e); };

 312}

 329/**

 330 * promises to delete a file or folder and return the file or folder name.

 331 * @param thePath the path to write

 332 * @return promise to provide the name of the removed file.

 333 */

 334export async function remove(thePath:string):Promise<string> {

 335    const dir:boolean = await isDirectory(thePath);

 336 return await new Promise((resolve, reject) => {

 337        dir? fs.rmdir(thePath, (e:any) => (e? reject(e) : resolve(thePath)))

 338           : fs.unlink(thePath, (e:any) => (e? reject(e) : resolve(thePath)));

 339 });

 340}

 342/**

 343 * promises to delete a file or folder and return the file or folder name.

 344 * @param thePath the path to write

 345 * @return promise to provide the list of names of the removed files.

 346 */

 347export async function removeAll(thePath:string):Promise<string[]> {

 348    const removed = [];

 349    const dir:boolean = await isDirectory(thePath);

 350    if (dir) {

 351        const list = await readDir(thePath);

 352        await Promise.all(list.map(async i => removed.push(...await removeAll(`${thePath}/${i}`))));

 353    }

 354    removed.push(await remove(thePath));

 355    return removed;

 356}
