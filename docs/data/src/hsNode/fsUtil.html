  45function stat(thePath:string):Promise {
  46 return Promise.resolve(thePath)
  47        .then(realPath)
  48 .then(thePath => new Promise((resolve:(value:Stats)=>void, reject) => {
  49 fs.stat(thePath, (err:any, stats:Stats) => {
  50 if(err) { reject(err); } // reject is hard to test: realpath throws an error before stat can.
  51 else { 
  52 stats.path = thePath;
  53 resolve(stats); 
  54 }
  55 });
  56 }));
  57}

  59function lstat(thePath:string) {
  60 return Promise.resolve(thePath)
  61 .then(path.normalize)
  62 .then(thePath => new Promise((resolve, reject) => {
  63 fs.lstat(thePath, (err:any, stats:any) => {
  64 if(err) { reject(err); }
  65 else    { 
  66 stats.path = thePath;
  67 resolve(stats); 
  68 }
  69 });
  70 }));
  71}

  81/**
  82 * determines the canonical path for `thePath`, resolving all symbolic links and '../'in the path.
  83 * @param thePath the path to check
  84 * @return promise to provide the real canonical system path.
  85 */
  86export function realPath(thePath:string):Promise {
  87 return new Promise((resolve:(path:string)=>void, reject:(err:any)=>void) => {
  88 fs.realpath(thePath, (err:any, resolvedPath:string) => err? reject(err) : resolve(resolvedPath) );
  89    })
  90    .catch(error);

 137/**
 138 * creates any missing directories in `thePath` and promises to return the path name.
 139 * Characters after the last `/` in `thePath` will be interpreted as a filename, hence no directory willbe created form them.
 140 * Terminate `thePath` with a final `/` to indicate that all parts should be created.
 141 * FInally, for precaution `mkdirs` only creates directories within the current working directory.
 142 * @param thePath the path to check
 143 * @return promise to provide the path name
 144 */ 
 145export function mkdirs(thePath:string):Promise {    
 146    const p = path.normalize(path.resolve(process.cwd(),thePath));
 147    if (p.indexOf(process.cwd())===0) { // --> thePath is local to current working directory
 148        const r = p.substr(process.cwd().length+1);
 149        let dirs = r.split('/');
 150        // create complete successive subdirs from the split
 151        dirs = dirs.map((dir, i) => './'+dirs.slice(0,i+1).join('/'));
 152        // setup the execution calls for each dir:
 153        const tasks = dirs.map(dir => () => isDirectory(dir)
 154            .then((exists) => new Promise((resolve:(dir:boolean)=>void, reject) => {
 155                if (exists) { 
 156                    log.debug(`dir exists: '${dir}'`);
 157                    resolve(true);
 158                } else { 
 159                    log.debug(`creating dir: '${dir}'`);
 160                    fs.mkdir(dir, (err:any) => {
 161                        if(!err) { resolve(true); }
 162                        else if (err.toString().match(/Error: EEXIST: file already exists/)) {
 163                            log.debug(`mkdirs: directory already exists: ${dir}`);
 164                            resolve(true);
 165                        } else {
 166                            reject(err);
 167                        }
 168                    });
 169                }
 170            }))
 171        );
 172        // serialize the directory creation: 
 173        return promiseChain(tasks).then((res:boolean[]) => {
 174            res.map((r, i) => {
 175                if (r) { return true; }
 176                throw `mkdir failed for ${dirs[i]}: ${r}`;
 177            });
 178            return dirs[dirs.length-1];
 179        });
 180    } else {
 181        return Promise.reject(`target '${p}' not inside working directory '${process.cwd()}'`);
 182    }
 183}

 190export function readDir(thePath:string):Promise {

 192 .then(realPath)
 193 .then(thePath => new Promise((resolve:(files:any)=>void, reject:(err:any)=>void) => {
 194 fs.readdir(thePath, (err:any, files:any) =>  {
 195 if(err) { reject(err); }

 197 files.path = thePath;
 198 resolve(files);

 200 });
 201 }))
 202        .catch(error);

 212export function readFile(thePath:string, isText=true):Promise {
 213 return new Promise((resolve:(data:any)=>void, reject:(err:any)=>void) => {
 214 let encoding = isText? 'utf8' : {};
 215 fs.readFile(thePath, encoding, (err:any, data:any) => {
 216            if (err) { reject(err); }

 218 });
 219 })
 220    .catch(error);

 244/**
 245 * writes a file either as binary or text and promises to return the file name.
 246 * @param thePath the path to write to
 247 * @param content the content to write
 248 * @param isText `true`|`false` if file should be read as `utf8`|binary 
 249 * @return promise to provide the file name if successful.
 250 */
 251export function writeFile(thePath:string, content:string, isText:boolean=true):Promise {
 252 return new Promise((resolve, reject) => {
 253        var encoding:any = isText? 'utf8' : {encoding: null};
 254        mkdirs(path.dirname(thePath))
 255        .then(() => fs.writeFile(thePath, content, encoding, (err:any) => 
 256            err? reject(err) : resolve(thePath))
 257        ).catch(err => log.error(`mkdirs failed in writeFile for '${thePath}': ${err}`));
 258 })
 259    .catch(error);
 260};

 262/**
 263 * writes content to a file as a stream and promises to return the file name.
 264 * @param thePath the path to write to
 265 * @param content the content to write
 266 * @return promise to provide the file name if successful.
 267 */
 268export function writeStream(thePath:string, content:string):Promise {
 269 return new Promise((resolve, reject) => {
 270        let s = fs.createWriteStream(thePath, {flags:'w', mode:0o666});
 271        s.on('error', (src:any) => reject(src));
 272        s.write(content, 'binary', () => resolve(thePath));
 273        s.end();
 274 })
 275    .catch(error);
 276}

 308export function appendFile(thePath:string, content:string, isText:boolean=true):Promise {
 309 return new Promise((resolve, reject) => {
 310 var encoding:any = isText? 'utf8' : {encoding: null};
 311        fs.appendFile(thePath, content, encoding, (err:any) => err? reject(err) : resolve(thePath));

 313    .then(realPath)
 314    .catch(error);

 317/**
 318 * promises to delete a file or folder and return the file or folder name.
 319 * @param thePath the path to write
 320 * @return promise to provide the name of the removed file.
 321 */
 322export function remove(thePath:string):Promise {
 323 return new Promise((resolve:(path:string)=>void, reject:(err:any)=>void) => {
 324        isDirectory(thePath).then((dir:boolean) => {
 325            dir? fs.rmdir(thePath, (e:any) => (e? reject(e) : resolve(thePath)))
 326               : fs.unlink(thePath, (e:any) => (e? reject(e) : resolve(thePath)));
 327        });
 328 })
 329    .catch(error);
